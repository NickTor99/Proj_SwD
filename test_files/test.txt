def operazione(integer op | float a | float b): float{
    if op = 1 then{
        return a + b;
    }
    if op = 2 then{
        var result << 0.0;
        while a >= 1 loop{
            a << a-1;
            result << result + b;
        }
        return result + (a*b);
    }
    if op = 4 then{
        return a^b;
    }
}

def fibonacci(integer a): integer{
    integer risultato << 0;
    if a = 0 then{
        return 0;
    }
    if a = 1 then{
        return 1;
    }
    return fibonacci(a-1) + fibonacci(a-2);
}

def divIntera(float a | float b | out string size): integer{
    integer result << a/b;
    if result >= setSize then{
        size << "grande";
    }
    return result;
}
